# What it does
Replace token from file to another file

# Installing
1. nvm install lts/fermium
2. nvm use lts/fermium
1. npm install -g
2. npm install -g pkg
3. pkg .

# Using
./dist/replacetoken ../envdesc-cli/envdesc-cli-2.3.0/altdev.json deployment/helm/imec/values.yaml -o toto.yaml
